                let iconClass = 'ph-fill ph-activity-heartbeat';
                let bgClass = 'bg-slate-100 text-slate-700 border-slate-200';

                if (part.includes("BASKETBOL")) {
                    iconClass = 'ph-fill ph-basketball';
                    bgClass = 'bg-orange-50 text-orange-600 border-orange-200/50';
                } else if (part.includes("VOLEYBOL")) {
                    iconClass = 'ph-fill ph-volleyball';
                    bgClass = 'bg-emerald-50 text-emerald-600 border-emerald-200/50';
                } else if (part.includes("CİMNASTİK")) {
                    iconClass = 'ph-fill ph-sparkles';
                    bgClass = 'bg-purple-50 text-purple-600 border-purple-200/50';
                } else if (part.includes("YÜZME")) {
                    iconClass = 'ph-fill ph-waves';

        // Tarih Sütun Başlıklarını Dinamik ve Gerçek Sütun (TH) Olarak Ekle (Hizalama Sorunu Çözümü)
        function buildMatrixHeaders() {
            const headerRow = document.getElementById('matrix-header-row');
            
            // Mevcut tarih hücreleri varsa temizle (Sabit kolonlardan sonrakileri uçur)
            headerRow.querySelectorAll('.date-th-cell').forEach(el => el.remove());

            timelineDates.forEach(date => {
                const headerStyle = date.isWeekend 
                    ? 'bg-amber-50 text-amber-950 border-amber-200/40' 
                    : 'bg-slate-50 text-slate-800 border-slate-200/40';

                const dayNameColor = date.isWeekend ? 'text-amber-600' : 'text-slate-400';

                const th = document.createElement('th');
                th.className = `date-th-cell w-10 min-w-[40px] border-r border-b border-slate-200 text-center py-2.5 select-none ${headerStyle}`;
                th.innerHTML = `
                    <div class="flex flex-col items-center justify-center">
                        <span class="text-[8px] font-extrabold text-slate-400 uppercase tracking-widest leading-none">${date.monthName}</span>
                        <span class="text-xs font-black tracking-tight my-0.5 leading-none">${date.day}</span>
                        <span class="text-[9px] font-bold ${dayNameColor} leading-none">${date.weekday}</span>
                    </div>
                `;
                headerRow.appendChild(th);
            });
        }

        // Satırları Oluştur (Branş Rozetleri ve Sarı Hücre Alanları ile Birlikte)
        function renderMatrixRows(data) {
            const tbody = document.getElementById('matrix-rows');
            tbody.innerHTML = '';

            data.forEach((row, rowIndex) => {
                const tr = document.createElement('tr');
                tr.className = "hover:bg-slate-50/50 transition-colors";

                // 1. Sıra No (Sticky left-0)
                const tdNum = document.createElement('td');
                tdNum.className = "sticky-col w-12 min-w-[48px] px-3 py-3 text-center font-bold text-slate-400 border-r border-b border-slate-200 bg-slate-100 sticky left-0 z-20";
                tdNum.textContent = rowIndex + 1;
                tr.appendChild(tdNum);

                // 2. Branş İkonları (Sticky left-12)
                const tdBranch = document.createElement('td');
                tdBranch.className = "sticky-col w-20 min-w-[80px] px-2 py-3 border-r border-b border-slate-200 bg-[#FFFDE7] sticky left-12 z-20 text-center";
                tdBranch.innerHTML = getBranchOnlyIcons(row.branch);
                tr.appendChild(tdBranch);

                // 3. Instagram Hesabı (Sticky left-[132px])
                const tdInstagram = document.createElement('td');
                tdInstagram.className = "sticky-col w-64 min-w-[256px] px-4 py-3 font-semibold text-blue-600 underline border-r border-b border-slate-200 bg-white sticky left-[132px] z-20 truncate";
                tdInstagram.innerHTML = `
                    <a href="https://instagram.com/${row.link}" target="_blank" class="flex items-center gap-1.5 hover:text-blue-800 transition-colors">
                        <i class="ph ph-instagram-logo text-base text-pink-500 shrink-0"></i>
                        <span class="truncate">${row.link}</span>
                    </a>
                `;
                tr.appendChild(tdInstagram);

                // 4. Kulüp / Takım Adı (Sticky left-[388px])
                const tdClub = document.createElement('td');
                tdClub.className = "sticky-col w-64 min-w-[256px] px-4 py-3 font-extrabold text-[12px] border-r border-b border-slate-200 bg-[#FFFDE7] sticky left-[388px] z-20 truncate";
                tdClub.textContent = row.club;
                tr.appendChild(tdClub);

                // 5. Tarih Matris Hücreleri (Kaydırılabilir alan)
                row.planStates.forEach((isPlanned, dateIndex) => {
                    const isWeekend = timelineDates[dateIndex].isWeekend;
                    const tdCell = document.createElement('td');
                    tdCell.className = "w-10 min-w-[40px] border-r border-b border-slate-200/80 transition-all text-center p-0";
                    
                    let bgClass = 'bg-white hover:bg-slate-100/50 cursor-pointer';
                    let dotHTML = '';
                    
                    if (isPlanned) {
                        bgClass = 'bg-[#3B82F6] hover:bg-blue-600 transition-colors cursor-pointer shadow-sm';
                        dotHTML = `<div class="w-1.5 h-1.5 rounded-full bg-white mx-auto shadow-sm"></div>`;
                    } else if (isWeekend) {
                        bgClass = 'bg-amber-50/20 hover:bg-amber-100/30 cursor-pointer';
                    }

                    tdCell.className += ` ${bgClass}`;
                    tdCell.innerHTML = `<div class="h-11 flex items-center justify-center">${dotHTML}</div>`;
                    tdCell.onclick = () => toggleCell(rowIndex, dateIndex);
                    
                    tr.appendChild(tdCell);
                });

                tbody.appendChild(tr);
            });
        }

        // Hücre Aktif/Pasif Yapıcı (Görsel Matris Etkileşimi)
        function toggleCell(rowIndex, dateIndex) {
            matrixData[rowIndex].planStates[dateIndex] = !matrixData[rowIndex].planStates[dateIndex];
            renderMatrixRows(matrixData);
        }

        // Matris Arama Filtresi
        function filterMatrix() {
            const term = document.getElementById('matrix-search').value.toLowerCase();
            const filtered = matrixData.filter(row => 
                row.club.toLowerCase().includes(term) || 
                row.branch.toLowerCase().includes(term) || 
                row.link.toLowerCase().includes(term)
            );
            renderMatrixRows(filtered);
        }

        function saveMatrixData() {
            const btn = document.getElementById('btn-save-matrix');
            const icon = document.getElementById('save-icon');
            const text = document.getElementById('save-text');
            const loader = document.getElementById('save-loader');

            btn.classList.add('opacity-90', 'pointer-events-none');
            icon.classList.add('hidden');
            loader.classList.remove('hidden');
            text.textContent = "Gönderiliyor...";

            setTimeout(() => {
                btn.classList.remove('opacity-90', 'pointer-events-none');
                icon.classList.remove('hidden');
                loader.classList.add('hidden');
                text.textContent = "Sheet'e Kaydet";

                showToast("Matris verileri Google Sheet ile senkronize edildi!");

                logsData.unshift({
                    team: "Tüm Kulüpler",
                    action: "Matris planlaması Google Sheet'e yazdırıldı.",
                    time: new Date().toLocaleTimeString('tr-TR', { hour: '2-digit', minute: '2-digit' }),
                    date: "27 Mayıs 2026"
                });
                renderLogs();

            }, 1200);
        }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            document.getElementById('toast-message').textContent = msg;
            
            toast.classList.remove('translate-y-10', 'opacity-0');
            toast.classList.add('translate-y-0', 'opacity-100');
            
            setTimeout(() => {
                toast.classList.remove('translate-y-0', 'opacity-100');
                toast.classList.add('translate-y-10', 'opacity-0');
            }, 3000);
        }
